60 public static double[][] multiplyMatrices(double a[][], double b[][]) {
61 if (a.length == 0) {
62 return new double[0][0];
63 }
64 if (a[0].length != b.length) {
65 return null;
66 }
67 int n = a[0].length;

72 for (int j = 0; j < p; j++) {

244 public static boolean thereIsPath(int[][] distanceMatrix, int node) {
245 HashSet<Integer> visits = new HashSet<>();
246 HashSet<Integer> noVisits = new HashSet<>();
247 for (int i = 0; i < distanceMatrix.length; i++) {
248 if (i != node) {
249 noVisits.add(i);
250 }
251 }
252 visits.add(node);
253 while (!visits.isEmpty()) {
254 Iterator<Integer> it = visits.iterator();
255 int toCheck = it.next();
256 visits.remove(toCheck);
257 for (int i = 0; i < distanceMatrix.length; i++) {
258 if (toCheck != i && distanceMatrix[toCheck][i] != Integer.MAX_VALUE) {
259 if (i == node) {
260 return true;
261 }
262 if (noVisits.contains(i)) {
263 noVisits.remove(i);
264 visits.add(i);
265 }
266 }
267 }
268 }
269 return false;
270 }
